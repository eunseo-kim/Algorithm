def solution():
    infix = list(input())
    stack = []
    answer = []
    # ")"ì˜ ìš°ì„ ìˆœìœ„ > */ì˜ ìš°ì„ ìˆœìœ„ > +-ì˜ ìš°ì„ ìˆœìœ„ > "("ì˜ ìš°ì„ ìˆœìœ„
    priority = {")": 3, "*": 2, "/": 2, "+": 1, "-": 1, "(": 0}

    for t in infix:
        if t.isalpha():
            answer.append(t)
        elif t == "(":
            stack.append(t)
        elif t == ")":
            while stack and stack[-1] != "(":
                answer.append(stack.pop())
            stack.pop()
        else:
            while stack and priority[stack[-1]] >= priority[t]:
                answer.append(stack.pop())
            stack.append(t)

    while stack:
        answer.append(stack.pop())

    return "".join(answer)


print(solution())
